% Formatting title and subtitle.
% https://tex.stackexchange.com/questions/120516/set-frame-subtitle-size
\setbeamertemplate{frametitle}{%
\vspace*{5mm}%
\linespread{1.0}%
\color{black}%
\bfseries%
% Invisible vertical space added so the text starts in the same
% vertical position always (even if not using taller characters such
% as "g" or "y").
% https://tex.stackexchange.com/questions/109816/automatic-frame-titles-subtitles-with-condition
\ifx\insertframetitle\empty%
\else%
\Large\insertframetitle\vphantom{AQgpqy}\par%
\fi%
\ifx\insertframesubtitle\empty%
\else%
\vspace*{2pt}%
\scriptsize\insertframesubtitle\vphantom{AQgpqy}\par%
%\large\insertframesubtitle\vphantom{AQgpqy}\par%
\fi%
\vspace*{-2.2mm}%
\CurrentProgressBarIs%
\vspace*{2mm}%
}

% To remove the navigation symbols.
\setbeamertemplate{navigation symbols}{}

\setbeamersize{text margin left=10mm}
\setbeamersize{text margin right=10mm}
\setbeamersize{sidebar width left=0mm}
\setbeamersize{sidebar width right=0mm}

\setbeamercolor{structure}{fg=black}

% Change itemize bullets.
% https://tex.stackexchange.com/questions/446524/beamer-controlling-the-size-of-custom-bullets
% https://tex.stackexchange.com/questions/4860/customizing-itemize-in-beamer
% https://tex.stackexchange.com/questions/11168/change-bullet-style-formatting-in-beamer

\setbeamertemplate{itemize subitem}{\raisebox{0.15em}{\scalebox{0.55}{$\bullet$}}}
\setbeamertemplate{itemize subsubitem}{\raisebox{0.15em}{\scalebox{0.50}{$\bullet$}}}
